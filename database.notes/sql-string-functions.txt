-- LOCATE(), INSTR() --

-- Find String Positions:


//MySQL LOCATE Function:

 LOCATE(): 

 Syntex: 

    LOCATE(substr,str)
    LOCATE(substr,str,pos) 


Examples: 

      SELECT LOCATE('a','An Apple',2)      
      SELECT LOCATE('a', NAME) FROM products;
      



//MySQL INSTR Function:

 Syntex: 

       INSTR(str,substr)

Examples: 

	SELECT INSTR('An Apple','a');
	SELECT INSTR(NAME, 'a') FROM products;
