Example 01:   In JavaScript

//Validate gmail.com

const string = "ajay_sks.001@gmail.com";
let pattern = /^[0-9a-z._-]+@gmail.com$/;

const m = string.match(pattern);

document.write(m);



//In Condition


if (string.match(pattern)) {
    document.write("Valid");
}else{
    document.write("Invalid");
}
