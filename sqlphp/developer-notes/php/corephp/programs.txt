Reversing a number: 

//Using while loop
$num = 23456;  
$revnum = 0;  
for ($num > 1)  
{  
$rem = $num % 10;  
$revnum = ($revnum * 10) + $rem;  
$num = ($num / 10);   
}  
echo "Reverse number of 23456 is: $revnum";  

--OR-- 

//Using for loop

$num = 23456;  
$revnum = 0;  
for ($num=23456;$num > 1;$num /=10)  
{  
$rem = $num % 10;  
$revnum = ($revnum * 10) + $rem;  
//$num = ($num / 10);   
}  
echo "Reverse number of 23456 is: $revnum";
